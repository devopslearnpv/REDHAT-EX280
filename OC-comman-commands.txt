Day 2 â€“ Application Deployment
# Deploy app from image

Day 5 â€“ Security & Quotas
# Service Account
oc create sa mysa

oc describe sa mysa

# Run pod with SA
oc run pod1 --image=busybox --dry-run=client -o yaml --overrides='{"spec": {"serviceAccountName": "mysa"}}'

# Assign SCC
oc adm policy add-scc-to-user anyuid -z mysa -n dev

# Resource Quota
oc create quota myquota --hard=cpu=2,memory=4Gi,pods=10

# LimitRange
oc create limitrange mem-limit --limits=memory=512Mi,1Gi,cpu=500m,1
